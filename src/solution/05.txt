import { useState } from "react";

export default function Notifications() {
  const [notifications, setNotifications] = useState(true);
  const [clickCount, setClickCount] = useState(0);
  const [lastState, setLastState] = useState("Be");

  const handleToggle = () => {
    setNotifications((prev) => {
      setLastState(prev ? "Ki" : "Be");
      return !prev;
    });

    setClickCount((prev) => prev + 1);
  };

  const handleReset = () => {
    setNotifications(true);
    setClickCount(0);
    setLastState("Be");
  };

  return (
    <div className="card">
      <p className="status">
        {notifications
          ? "√ârtes√≠t√©sek be vannak kapcsolva"
          : "√ârtes√≠t√©sek ki vannak kapcsolva"}
      </p>

      <div className="meta">
        <p className="meta-item">Kattint√°sok sz√°ma: {clickCount}</p>
        <p className="meta-item">Utols√≥ √°llapot: {lastState}</p>
      </div>

      <div className="actions">
        <button className="toggle" onClick={handleToggle}>
          {notifications ? "Kikapcsol√°s" : "Bekapcsol√°s"}
        </button>

        <button className="reset" onClick={handleReset}>
          Vissza√°ll√≠t√°s
        </button>
      </div>

      {clickCount >= 5 && (
        <p className="warning">‚ö†Ô∏è T√∫l sok kattint√°s!</p>
      )}
    </div>
  );
}
